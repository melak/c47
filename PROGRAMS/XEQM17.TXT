XEQLBL 17 PLOT_LP    //RE-BINDS THE TEXT TO THE SOFTMENU 17, i.e. G[FN5]

//PROGRAM TO PLOT 100 RANDOM NUMBERS ON A LINE
// Y = 0.1 X + 3 + RAN#, for X starting at -10 to 20 

CLSUM              //USE CLSUM AS THE INTERPRETER CANNOT HANDLE CLΣ
TICKS              //GET CURRENT TIME IN 0.1 SECOND UNITS
STO "01"        
        
"20"               //LOOP COUNTER STARTS AT 20 AND DECREMENTS TO 0
STO "00"           //LOOP COUNTER
STO "02"           //LOOP COUNTER START VALUE

LBL M1             //CREATE LOOPING LABEL

RCL "02" 
RCL "00" -         //0-40
"20" / "30" * 
"10" -             //-10 to 20
ENTER              //DUPLICATE TO Y
"0.1" * "3" + 
RAN# +             //Y IN X NOW
X<>Y               //Y CORRECT X CORRECT
SUM+               //USE SUM+ AS THE INTERPRETER CANNOT HANDLE Σ+

DSZ "00"           //STANDARD DSZ, DEC AND SKIP NEXT COMMAND IF RESULT WAS ZERO
GTO M1
        
TICKS              //GET CLOCK STATUS
RCL "01"           //GET OLD CLOCK STATUS
-                  //SUBTRACT
"10"               //DIVIDE BY 10 TO GET SECONDS
/        
        
PLOT
